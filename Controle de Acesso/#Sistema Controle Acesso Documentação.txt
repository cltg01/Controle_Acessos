Modelagem do Banco de Dados;

CREATE DATABASE sistema;

CREATE TABLE tabela_usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nome_usuario VARCHAR(50),
    email_usuario VARCHAR(50),
    senha_usuario VARCHAR(255)
);

CREATE TABLE tabela_credenciado (
    id_credenciado INT AUTO_INCREMENT PRIMARY KEY,
    nome_credenciado VARCHAR(50),
    cpf_credenciado VARCHAR(11) UNIQUE,
    telefone_credenciado VARCHAR(20),
    email_credenciado VARCHAR(50),
    tipo_credenciado ENUM('Fornecedor', 'Cliente', 'Prestador de Serviço', 'Terceirizado', 'Liderança', 'Outro'),
    empresa_organizacao VARCHAR(50),
    foto_credenciado LONGBLOB,
    observacoes_credenciado TEXT
);

CREATE TABLE tabela_acesso (
    id_acesso INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT,
    id_credenciado INT,
    motivo_acesso TEXT,
    data_hora_entrada DATETIME NOT NULL,
    data_hora_saida DATETIME,
    status_acesso ENUM('Em andamento', 'Concluído', 'Cancelado') DEFAULT 'Em andamento',
    FOREIGN KEY (id_usuario) REFERENCES tabela_usuario(id_usuario),
    FOREIGN KEY (id_credenciado) REFERENCES tabela_credenciado(id_credenciado)
);
